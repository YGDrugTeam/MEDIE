
azure 로그인
az login

azure 그룹생성
az group create --name medichubs-rg --location eastus2

ACR (Azure Container Registry) 만들기
az acr create `
  --resource-group final_4 `
  --name medichubsacr `
  --sku Basic `
  --admin-enabled true
  ##만들고서 loginServer 복사하기 
  "medichubsacr.azurecr.io"


acr 로그인
az acr login --name medichubsacr


docker build -t medichubsacr.azurecr.io/medichubs-backend:latest .

acr 로 푸시
docker tag medichubs-backend:latest medichubsacr.azurecr.io/medichubs-backend:latest
docker push medichubsacr.azurecr.io/medichubs-backend:latest

4-1️⃣ App Service Plan 만들기 (Linux)
az appservice plan create `
  --name medichubs-plan `
  --resource-group final_4 `
  --location westus2 `
  --sku B1 `
  --is-linux


4-2️⃣ App Service(Web App) 생성 (Docker 이미지 연결)
az webapp create `
  --resource-group final_4 `
  --plan medichubs-plan `
  --name medichubs-backend `
  --deployment-container-image-name medichubsacr.azurecr.io/medichubs-backend:latest

  4-3️⃣ App Service ↔ ACR 인증 연결 (중요🔥)
  az webapp config container set `
  --name medichubs-backend `
  --resource-group final_4 `
  --docker-custom-image-name medichubsacr.azurecr.io/medichubs-backend:latest `
  --docker-registry-server-url https://medichubsacr.azurecr.io `
  --docker-registry-server-user $(az acr credential show -n medichubsacr --query username -o tsv) `
  --docker-registry-server-password $(az acr credential show -n medichubsacr --query passwords[0].value -o tsv)



4-4️⃣ 포트 설정 (FastAPI / Uvicorn 필수)

네 서버는 8000번 포트 쓰고 있으니까
  az webapp config appsettings set `
  --name medichubs-backend `
  --resource-group final_4 `
  --settings WEBSITES_PORT=8000


4-5️⃣ 서버 재시작
  az webapp restart `
  --name medichubs-backend `
  --resource-group final_4
